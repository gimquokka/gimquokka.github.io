from collections import deque


def bfs(graph, start, visited):
    q = deque([start])
    visited[start] = True
    # 1. Basic
    while q:
        v = q.popleft()
        for i in graph[v]:
            if not visited[i]:
                q.append(i)
                visited[i] = True
